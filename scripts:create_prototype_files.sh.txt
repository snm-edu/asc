import * as THREE from "https://unpkg.com/three@0.165.0/build/three.module.js";
import { OrbitControls } from "https://unpkg.com/three@0.165.0/examples/jsm/controls/OrbitControls.js";

const viewport = document.getElementById("viewport");
const partName = document.getElementById("partName");
const partNameEn = document.getElementById("partNameEn");
const partDescription = document.getElementById("partDescription");
const domainFilter = document.getElementById("domainFilter");
const layerFilter = document.getElementById("layerFilter");
const toggleEn = document.getElementById("toggleEn");
const quizButton = document.getElementById("quizButton");
const quizArea = document.getElementById("quizArea");
const quizPrompt = document.getElementById("quizPrompt");
const quizOptions = document.getElementById("quizOptions");
const learningTime = document.getElementById("learningTime");
const quizScore = document.getElementById("quizScore");

const scene = new THREE.Scene();
scene.background = new THREE.Color("#dce9ff");

const camera = new THREE.PerspectiveCamera(45, 1, 0.1, 100);
camera.position.set(1.8, 1.4, 2.2);

const renderer = new THREE.WebGLRenderer({ antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
viewport.appendChild(renderer.domElement);

const controls = new OrbitControls(camera, renderer.domElement);
controls.enableDamping = true;
controls.target.set(0, 1.1, 0);
controls.rotateSpeed = 0.9;
controls.zoomSpeed = 1.0;
controls.panSpeed = 0.9;
controls.touches.ONE = THREE.TOUCH.ROTATE;
controls.touches.TWO = THREE.TOUCH.DOLLY_PAN;

const hemiLight = new THREE.HemisphereLight(0xffffff, 0x7482a0, 1.4);
scene.add(hemiLight);
const dirLight = new THREE.DirectionalLight(0xffffff, 1);
dirLight.position.set(3, 5, 4);
scene.add(dirLight);

const bodyRoot = new THREE.Group();
scene.add(bodyRoot);

const referenceBody = new THREE.Mesh(
  new THREE.CapsuleGeometry(0.35, 1.2, 10, 18),
  new THREE.MeshStandardMaterial({ color: "#9eb8de", transparent: true, opacity: 0.35 })
);
referenceBody.position.y = 0.95;
bodyRoot.add(referenceBody);

const raycaster = new THREE.Raycaster();
const pointer = new THREE.Vector2();
let anatomyParts = [];
let selectedMesh = null;
let showEnglish = false;
let startTime = Date.now();
let quizState = { active: false, total: 0, correct: 0, answerId: null };

const floatingLabel = document.createElement("div");
floatingLabel.className = "floating-label hidden";
viewport.appendChild(floatingLabel);

function createGeometry(shape) {
  if (shape === "sphere") {
    return new THREE.SphereGeometry(0.5, 32, 24);
  }
  if (shape === "box") {
    return new THREE.BoxGeometry(1, 1, 1);
  }
  return new THREE.CapsuleGeometry(0.5, 0.8, 8, 16);
}

function loadParts() {
  return fetch("./data/anatomy_parts.json").then((res) => res.json());
}

function buildMeshes(parts) {
  for (const part of parts) {
    const geometry = createGeometry(part.shape);
    geometry.scale(...part.scale);
    const material = new THREE.MeshStandardMaterial({ color: part.color });
    const mesh = new THREE.Mesh(geometry, material);
    mesh.position.set(...part.position);
    mesh.userData.part = part;
    bodyRoot.add(mesh);
    anatomyParts.push(mesh);
  }
}

function applyFilters() {
  const domain = domainFilter.value;
  const layer = layerFilter.value;

  anatomyParts.forEach((mesh) => {
    const part = mesh.userData.part;
    const domainMatch = domain === "all" || part.domains.includes(domain);
    const layerMatch = layer === "all" || part.layer === layer;
    mesh.visible = domainMatch && layerMatch;
  });
}

function clearSelection() {
  if (selectedMesh) {
    selectedMesh.material.emissive?.setHex(0x000000);
  }
  selectedMesh = null;
  floatingLabel.classList.add("hidden");
}

function renderPartInfo(part) {
  partName.textContent = part.nameJa;
  partNameEn.textContent = showEnglish ? part.nameEn : "";
  partDescription.textContent = part.description;
}

function selectMesh(mesh, point) {
  clearSelection();
  selectedMesh = mesh;
  selectedMesh.material.emissive = new THREE.Color("#2f5fff");

  const part = mesh.userData.part;
  renderPartInfo(part);

  const labelText = showEnglish ? `${part.nameJa} / ${part.nameEn}` : part.nameJa;
  floatingLabel.textContent = labelText;
  floatingLabel.classList.remove("hidden");

  const projected = point.clone().project(camera);
  const x = (projected.x * 0.5 + 0.5) * viewport.clientWidth;
  const y = (projected.y * -0.5 + 0.5) * viewport.clientHeight - 20;
  floatingLabel.style.left = `${x}px`;
  floatingLabel.style.top = `${y}px`;
}

function onPointerTap(event) {
  const rect = renderer.domElement.getBoundingClientRect();
  pointer.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
  pointer.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;

  raycaster.setFromCamera(pointer, camera);
  const intersects = raycaster.intersectObjects(anatomyParts.filter((m) => m.visible));

  if (intersects.length > 0) {
    selectMesh(intersects[0].object, intersects[0].point);

    if (quizState.active) {
      handleQuizAnswer(intersects[0].object.userData.part.id);
    }
  }
}

function updateLearningTime() {
  const minutes = Math.floor((Date.now() - startTime) / 60000);
  learningTime.textContent = `${minutes}分`;
}

function startQuiz() {
  const visibleParts = anatomyParts.filter((mesh) => mesh.visible).map((mesh) => mesh.userData.part);
  if (visibleParts.length < 2) {
    quizPrompt.textContent = "クイズ出題には表示中の部位が2つ以上必要です。";
    quizArea.classList.remove("hidden");
    return;
  }

  const answer = visibleParts[Math.floor(Math.random() * visibleParts.length)];
  const shuffled = [...visibleParts].sort(() => Math.random() - 0.5).slice(0, 4);
  if (!shuffled.find((part) => part.id === answer.id)) {
    shuffled[0] = answer;
  }

  quizState.active = true;
  quizState.answerId = answer.id;
  quizState.total += 1;

  quizPrompt.textContent = `「${answer.nameJa}」をタップするか、下から選択してください。`;
  quizOptions.innerHTML = "";

  shuffled.forEach((part) => {
    const btn = document.createElement("button");
    btn.className = "quiz-option";
    btn.textContent = part.nameJa;
    btn.addEventListener("click", () => handleQuizAnswer(part.id));
    quizOptions.appendChild(btn);
  });

  quizArea.classList.remove("hidden");
}

function handleQuizAnswer(selectedId) {
  if (!quizState.active) {
    return;
  }

  if (selectedId === quizState.answerId) {
    quizState.correct += 1;
    quizPrompt.textContent = "正解です！次の問題を開始できます。";
  } else {
    quizPrompt.textContent = "不正解です。もう一度クイズ開始を押してください。";
  }

  quizState.active = false;
  quizScore.textContent = `${Math.round((quizState.correct / quizState.total) * 100)}% (${quizState.correct}/${quizState.total})`;
}

function resize() {
  const width = viewport.clientWidth;
  const height = viewport.clientHeight;
  camera.aspect = width / height;
  camera.updateProjectionMatrix();
  renderer.setSize(width, height);
}

domainFilter.addEventListener("change", applyFilters);
layerFilter.addEventListener("change", applyFilters);
toggleEn.addEventListener("click", () => {
  showEnglish = !showEnglish;
  if (selectedMesh) {
    renderPartInfo(selectedMesh.userData.part);
  }
});
quizButton.addEventListener("click", startQuiz);
renderer.domElement.addEventListener("pointerup", onPointerTap);
window.addEventListener("resize", resize);

loadParts()
  .then((parts) => {
    buildMeshes(parts);
    applyFilters();
    resize();
  })
  .catch(() => {
    partName.textContent = "データ読み込みエラー";
    partDescription.textContent = "data/anatomy_parts.json が見つからないか、形式が正しくありません。";
  });

function animate() {
  requestAnimationFrame(animate);
  controls.update();
  updateLearningTime();
  renderer.render(scene, camera);
}

animate();
EOF

# Human Anatomy Atlas 2026 Prototype

iPad向け「ヒューマン・アナトミー・アトラス2026」の実装プロトタイプです。

## 起動方法

```bash
python3 -m http.server 4173
```

ブラウザで `http://localhost:4173` を開いてください。

## 実装済み

- 一体型3D解剖体（1つの人体モデル空間に全領域の部位を保持）
- 1本指相当: 回転 / タップ選択
- 2本指相当: 拡大縮小・平行移動
- 領域フィルタ・レイヤーフィルタ
- 部位ラベル（日本語・英語切替）
- クイズ機能（名称→部位）
- 学習時間・正答率の表示

## 備考

- 本実装はUI/操作の検証を主目的としたプロトタイプです。
- 医療教育での本運用には、実解剖に忠実な監修済み3Dデータへの差し替えが必要です。


## 作成ファイル

- `src/app.js`（3D処理本体）
- `src/styles.css`（UIスタイル）
- `data/anatomy_parts.json`（部位データ）
- `README.md`（起動手順）


## GitHub Pages公開手順（404対策）

1. このリポジトリを `main` に push する
2. GitHub の `Settings > Pages` で Source を `GitHub Actions` にする
3. `Actions` タブで `Deploy static site to GitHub Pages` が成功するまで待つ
4. `https://snm-edu.github.io/asc/` を開く

### 404が出る場合

- URLが `https://snm-edu.github.io/asc/` になっているか確認（`/asc/` が必要）
- Actions のデプロイが成功しているか確認
- 初回デプロイ直後は反映まで数分かかる場合あり
EOF

# ヒューマン・アナトミー・アトラス2026 iPadアプリ 実装指示書

## 1. 目的
札幌看護医療専門学校に入学する高校生が、看護師・視能訓練士・歯科衛生士・臨床工学技士で必要となる解剖学を、**一体型の高精度3D解剖体**を操作しながら学べるiPadアプリを実装する。

> 重要方針: 学習領域は職種別に分類表示できるが、3Dモデル自体は分離しない。  
> **1つの人体解剖モデルに全領域が統合されていることを必須要件**とする。

---

## 2. 対象ユーザー
- 札幌看護医療専門学校へ進学予定の高校生
- 解剖学初学者
- iPadユーザー（授業・自宅学習）

---

## 3. プラットフォーム・基本仕様
- 対応OS: iPadOS（最新 + 2世代前まで）
- 対応向き: 横向き固定（推奨）
- オフライン対応: 主要学習機能はオフライン利用可
- パフォーマンス目標:
  - 通常表示: 30fps以上
  - 部位選択反応: 200ms以内目標

---

## 4. 3Dモデル実装要件（最重要）

### 4.1 モデル構造
- 3Dモデルは**全身一体型**とし、以下を同一モデル内に保持すること。
  - 骨格
  - 筋
  - 神経
  - 血管
  - 臓器
  - 眼関連詳細
  - 口腔・歯科関連詳細
  - 心肺・腎など臨床工学関連ランドマーク

### 4.2 領域分類の扱い
- 看護師 / 視能訓練士 / 歯科衛生士 / 臨床工学技士 の分類は、
  - 表示フィルタ
  - 学習コース
  - クイズ出題範囲
 で切り替え可能にする。
- ただし分類切替時も、同じ3D解剖体を使い続けること（モデル差し替え禁止）。

### 4.3 正確性
- 実在解剖学に準拠した位置・形状・相対関係を維持
- 学科教員または医療監修者レビューを通過したモデルのみ採用

---

## 5. ジェスチャー操作要件
- **1本指ドラッグ**: モデル回転
- **2本指ピンチ**: 拡大/縮小
- **2本指ドラッグ**: 平行移動（パン）
- **1本指タップ**: 部位選択
  - 選択部位をハイライト表示
  - 部位名ラベルをポップアップ表示

---

## 6. 学習UI要件
### 6.1 画面構成
- 中央: 3D表示領域
- 上部: レイヤー切替・検索・領域フィルタ
- 下部: 部位情報（日本語名、英語名、説明）
- 側部: クイズモード切替、学習履歴

### 6.2 ラベル表示
- タップした部位に対して以下を表示
  - 日本語名称（デフォルト）
  - 英語名称（ON/OFF）
  - 簡易説明（高校生向け）

### 6.3 レイヤー
- 皮膚 / 筋 / 骨 / 神経 / 血管 / 臓器 を個別に表示ON/OFF
- レイヤー切替時にカメラ位置を維持し、連続学習を妨げない

---

## 7. 学習機能要件
- 名称学習モード
  - 任意部位をタップして名称確認
- クイズモード
  - 「この名称はどこ？」（部位選択）
  - 「この部位名は？」（名称選択）
- 進捗管理
  - 学習時間、正答率、苦手部位

---

## 8. データ設計要件
- 各部位にユニークIDを付与
- 部位情報テーブルに以下を保持
  - 部位ID
  - 日本語名
  - 英語名

name: Deploy static site to GitHub Pages

on:
  push:
    branches:
      - main
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

  group: pages

  deploy:
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Pages
        uses: actions/configure-pages@v5

      - name: Upload artifact
        uses: actions/upload-pages-artifact@v3
        with:
          path: .

      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4
